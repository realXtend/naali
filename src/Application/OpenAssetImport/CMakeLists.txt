# Define target name
init_target (OpenAssetImport)

file (GLOB H_FILES *.h)

set (SOURCE_FILES ${CPP_FILES} ${H_FILES})

build_library (${TARGET_NAME} STATIC ${SOURCE_FILES} ${MOC_SRCS})

link_modules (Framework Asset)
link_ogre()
link_package_assimp()
link_package(QT4)

SetupCompileFlags()

final_target ()
